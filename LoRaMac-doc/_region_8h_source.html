1460 /*
1461  * Rx window precise timing
1462  *
1463  * For more details please consult the following document, chapter 3.1.2.
1464  * https://www.semtech.com/uploads/documents/SX1272_settings_for_LoRaWAN_v2.0.pdf
1465  * or
1466  * https://www.semtech.com/uploads/documents/SX1276_settings_for_LoRaWAN_v2.0.pdf
1467  *
1468  * Downlink start: T = Tx + 1s (+/- 20 us)
1469  * |
1470  * TRxEarly | TRxLate
1471  * | | |
1472  * | | +---+---+---+---+---+---+---+---+
1473  * | | | Latest Rx window |
1474  * | | +---+---+---+---+---+---+---+---+
1475  * | | |
1476  * +---+---+---+---+---+---+---+---+
1477  * | Earliest Rx window |
1478  * +---+---+---+---+---+---+---+---+
1479  * |
1480  * +---+---+---+---+---+---+---+---+
1481  *Downlink preamble 8 symbols | | | | | | | | |
1482  * +---+---+---+---+---+---+---+---+
1483  *
1484  * Worst case Rx window timings
1485  *
1486  * TRxLate = DEFAULT_MIN_RX_SYMBOLS * tSymbol - RADIO_WAKEUP_TIME
1487  * TRxEarly = 8 - DEFAULT_MIN_RX_SYMBOLS * tSymbol - RxWindowTimeout - RADIO_WAKEUP_TIME
1488  *
1489  * TRxLate - TRxEarly = 2 * DEFAULT_SYSTEM_MAX_RX_ERROR
1490  *
1491  * RxOffset = ( TRxLate + TRxEarly ) / 2
1492  *
1493  * RxWindowTimeout = ( 2 * DEFAULT_MIN_RX_SYMBOLS - 8 ) * tSymbol + 2 * DEFAULT_SYSTEM_MAX_RX_ERROR
1494  * RxOffset = 4 * tSymbol - RxWindowTimeout / 2 - RADIO_WAKE_UP_TIME
1495  *
1496  * Minimal value of RxWindowTimeout must be 5 symbols which implies that the system always tolerates at least an error of 1.5 * tSymbol
1497  */
