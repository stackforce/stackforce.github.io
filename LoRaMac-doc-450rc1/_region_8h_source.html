1464 /*
1465  * Rx window precise timing
1466  *
1467  * For more details please consult the following document, chapter 3.1.2.
1468  * https://www.semtech.com/uploads/documents/SX1272_settings_for_LoRaWAN_v2.0.pdf
1469  * or
1470  * https://www.semtech.com/uploads/documents/SX1276_settings_for_LoRaWAN_v2.0.pdf
1471  *
1472  * Downlink start: T = Tx + 1s (+/- 20 us)
1473  * |
1474  * TRxEarly | TRxLate
1475  * | | |
1476  * | | +---+---+---+---+---+---+---+---+
1477  * | | | Latest Rx window |
1478  * | | +---+---+---+---+---+---+---+---+
1479  * | | |
1480  * +---+---+---+---+---+---+---+---+
1481  * | Earliest Rx window |
1482  * +---+---+---+---+---+---+---+---+
1483  * |
1484  * +---+---+---+---+---+---+---+---+
1485  *Downlink preamble 8 symbols | | | | | | | | |
1486  * +---+---+---+---+---+---+---+---+
1487  *
1488  * Worst case Rx window timings
1489  *
1490  * TRxLate = DEFAULT_MIN_RX_SYMBOLS * tSymbol - RADIO_WAKEUP_TIME
1491  * TRxEarly = 8 - DEFAULT_MIN_RX_SYMBOLS * tSymbol - RxWindowTimeout - RADIO_WAKEUP_TIME
1492  *
1493  * TRxLate - TRxEarly = 2 * DEFAULT_SYSTEM_MAX_RX_ERROR
1494  *
1495  * RxOffset = ( TRxLate + TRxEarly ) / 2
1496  *
1497  * RxWindowTimeout = ( 2 * DEFAULT_MIN_RX_SYMBOLS - 8 ) * tSymbol + 2 * DEFAULT_SYSTEM_MAX_RX_ERROR
1498  * RxOffset = 4 * tSymbol - RxWindowTimeout / 2 - RADIO_WAKE_UP_TIME
1499  *
1500  * Minimal value of RxWindowTimeout must be 5 symbols which implies that the system always tolerates at least an error of 1.5 * tSymbol
1501  */
